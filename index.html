⚙️ 자동 발송 스케줄러 설정 가이드
💡 매일 오전 9시 자동 검사 — 아래 Node.js 스크립트를 서버에 배포하고 cron 또는 GitHub Actions로 실행하면 인증서 만료일이 기준일(기본 30일) 이내로 들어올 때 담당자에게 자동 이메일이 발송됩니다.
// ssl-alert.js (Node.js · nodemailer 사용) const nodemailer = require('nodemailer'); const ALERT_DAYS = 30; // 만료 N일 전 알림 const SSL_DATA = [ { name: "가천대학교", expiry: "2026-08-03", manager: "김지선", email: "kimjiseon@company.com" }, { name: "건국대학교(서울)", expiry: "2026-07-05", manager: "김지선", email: "kimjiseon@company.com" }, { name: "건국대학교(글로컬)", expiry: "2026-04-05", manager: "권현지", email: "kwonhyunji@company.com" }, // ... (나머지 학교 데이터) ]; const transporter = nodemailer.createTransport({ host: "smtp.company.com", port: 587, auth: { user: "ssl-alert@company.com", pass: "PASSWORD" } }); async function checkAndAlert() { const today = new Date(); const alerts = SSL_DATA.filter(s => { const days = Math.ceil((new Date(s.expiry) - today) / 86400000); return days >= 0 && days <= ALERT_DAYS; }); const grouped = {}; alerts.forEach(s => { if (!grouped[s.email]) grouped[s.email] = { manager: s.manager, items: [] }; grouped[s.email].items.push(s); }); for (const [email, data] of Object.entries(grouped)) { await transporter.sendMail({ from: '"SSL 관리 시스템" ', to: email, subject: `[SSL 인증서 만료 알림] ${data.manager}님 담당 ${data.items.length}개교 갱신 필요`, html: `...` }); } } checkAndAlert().catch(console.error); // cron: 0 9 * * * /usr/bin/node /path/to/ssl-alert.js // GitHub Actions cron: '0 0 * * *' (UTC 0시 = KST 9시)
